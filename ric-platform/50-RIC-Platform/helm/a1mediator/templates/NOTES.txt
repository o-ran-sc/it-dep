  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "a1mediator.fullname" . }})

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "a1mediator.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
